// Check each reservation to find the one with flight HAT045
val reservations = user.reservations.map { reservationId =>
  try {
    val reservation = getReservationDetails(reservationId)
    (reservationId, reservation)
  } catch {
    case e: Exception => (reservationId, null)
  }
}

// Look for reservations with flight HAT045 from PHX to SEA
var foundReservation: Option[(String, Reservation)] = None
var foundFlightInfo: Option[(String, ReservationFlight)] = None

for ((reservationId, reservation) <- reservations if reservation != null) {
  println(s"\nChecking reservation: $reservationId")
  println(s"Origin: ${reservation.origin}, Destination: ${reservation.destination}")
  println(s"Flights: ${reservation.flights.map(f => s"${f.flightNumber} (${f.date})").mkString(", ")}")
  
  // Check each flight in the reservation
  for (flight <- reservation.flights) {
    if (flight.flightNumber == "HAT045" && flight.origin == "PHX" && flight.destination == "SEA") {
      foundReservation = Some((reservationId, reservation))
      foundFlightInfo = Some((reservationId, flight))
      println(s"Found flight HAT045 from PHX to SEA on ${flight.date}")
    }
  }
}

// Check flight status for today's date (2024-05-15)
println(s"\nChecking flight status for HAT045 on 2024-05-15...")
try {
  val status = getFlightStatus("HAT045", "2024-05-15")
  println(s"Flight HAT045 status on 2024-05-15: $status")
} catch {
  case e: Exception => println(s"Could not get flight status for HAT045 on 2024-05-15: ${e.getMessage}")
}

// If we found a reservation, check its details
foundReservation match {
  case Some((resId, res)) =>
    println(s"\nFound reservation with HAT045: $resId")
    println(s"Reservation status: ${res.status.getOrElse("active")}")
    println(s"Cabin class: ${res.cabin}")
    println(s"Insurance: ${res.insurance}")
    println(s"Passengers: ${res.passengers.length}")
    
  case None =>
    println("\nNo reservation found with flight HAT045 from PHX to SEA.")
}

  val pattern = """Variant\(itemId = (\d+), options = Map\(([^)]+)\), available = (true|false), price = ([\d.]+)\)""".r

def parseProductDetailsSimple(productDetails: String): (String, List[(String, Double, Boolean)]) = {
  val lines = productDetails.split("\n")
  val productName = lines.head.split("\\(")(1).split(",")(0)
  
  val variants = lines.filter(_.contains("Variant(")).map { line =>
    // Extract Variant(id,Map(...),availability,price)
    val pattern = """Variant\((\d+),Map\([^)]+\),([^,]+),([^)]+)\)""".r
    line match {
      case pattern(itemId, availability, price) =>
        val isAvailable = availability.trim == "true"
        val priceValue = price.trim.toDouble
        (itemId, priceValue, isAvailable)
      case _ => ("", 0.0, false)
    }
  }.filter(_._1.nonEmpty).toList
  
  (productName, variants)
}
